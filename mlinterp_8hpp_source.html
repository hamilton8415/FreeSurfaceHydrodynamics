54  template<typename Index>
55  void run(const Index * nd, Index n, Index * indices, T * weights)
56  {
57  // Must have at least one point per axis
58  assert(*nd > 0);
59 
60  const T x = xi[n];
61 
62  Index mid;
63  T weight;
64 
65  if (*nd == 1 || x <= xd[0]) {
66  // Data point is less than left boundary
67  mid = 0;
68  weight = 1.;
69  } else if (x >= xd[*nd - 1]) {
70  // Data point is greater than right boundary
71  mid = *nd - 2;
72  weight = 0.;
73  } else {
74  // Binary search to find tick
75  Index lo = 0, hi = *nd - 2;
76  mid = 0;
77  weight = 0.;
78  while (lo <= hi) {
79  mid = lo + (hi - lo) / 2;
80  if (x < xd[mid]) {
81  hi = mid - 1;
82  } else if (x >= xd[mid + 1]) {
83  lo = mid + 1;
84  } else {
85  weight = (xd[mid + 1] - x) / (xd[mid + 1] - xd[mid]);
86  break;
87  }
88  }
89  }
90 
91  *indices = mid;
92  *weights = weight;
93 
94  helper<T, Args...> & base = (*this);
95  base.run(nd + 1, n, indices + 1, weights + 1);
96  }

135 template<typename Order = natord, typename ... Args, typename T, typename Index>
136 static void interp(
137  const Index * nd, Index ni, const T * yd, T * yi,
138  Args... args)
139 {
140  // Cannot have a negative number of points
141  assert(ni >= 0);
142 
143  // Infer dimension from arguments
144  static_assert(sizeof...(Args) % 2 == 0, "needs 4+2*Dimension arguments");
145  constexpr Index Dimension = sizeof...(Args) / 2;
146 
147  // Compute 2^Dimension
148  constexpr Index Power = 1 << Dimension;
149 
150  // Unpack arguments
151  helper<T, Args...> h(args ...);
152 
153  // Perform interpolation for each point
154  Index indices[Dimension];
155  T weights[Dimension];
156  Index buffer[Dimension];
157  T factor;
158  for (Index n = 0; n < ni; ++n) {
159  yi[n] = 0.;
160  h.run(nd, n, indices, weights);
161  for (Index bitstr = 0; bitstr < Power; ++bitstr) {
162  factor = 1.;
163  for (Index i = 0; i < Dimension; ++i) {
164  if (bitstr & (1 << i)) {
165  buffer[i] = indices[i];
166  factor *= weights[i];
167  } else {
168  buffer[i] = indices[i] + 1;
169  factor *= 1 - weights[i];
170  }
171  }
172  if (factor > std::numeric_limits<T>::epsilon()) {
173  const Index k = Order::template mux<Index, Dimension>(nd, buffer);
174  yi[n] += factor * yd[k];
175  }
176  }
177  }
178 }
